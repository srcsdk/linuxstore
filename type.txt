ls : list everything in folder
cd <folder> : right click folder
cd .. : back out of folder
mkdir : make folder
touch <file.txt> : make file.txt
cp <file> <dest> : copy file to dest
mv <file> <dest> : move or rename file
rm <file> : delete file
rm -r <folder> : delete folder and everything in it
cat <file> : print file contents
grep <word> <file> : search for word in file
| : pipe output to next command
> : send output to file
>> : add output to end of file
chmod +x <file> : make file runnable
chmod 777 <file> : let everyone do everything to file
chown <user> <file> : change who owns file
sudo <command> : run as admin
su : switch to root
tar -xvf <file.tar> : extract tar
tar -czvf <name.tar.gz> <folder> : compress folder
wget <url> : download from url
curl <url> : get page content
ssh <user>@<ip> : connect to remote machine
scp <file> <user>@<ip>:<path> : copy file to remote machine
ps aux : show all running processes
kill <pid> : stop a process
top : live process monitor
df -h : show disk space
du -sh <folder> : show folder size
ifconfig : show network info
ping <ip> : check if something is alive
traceroute <ip> : show path to destination
ip addr : show ip addresses
ip route : show routing table
ss -tlnp : show listening ports
arp -n : show arp table
iw dev <iface> scan : scan wifi networks
iw dev <iface> info : show interface info
ip link set <iface> down/up : disable/enable interface
find / -perm -4000 : find suid binaries
stat <file> : show file details
awk '{print $1}' : print first column
sed 's/old/new/' : replace text
xargs -P <n> : run in parallel
nc <host> <port> : connect to port
echo > /dev/tcp/<host>/<port> : test if port open
tcpdump -i <iface> : capture packets
